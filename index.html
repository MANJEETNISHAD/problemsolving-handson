question:4 Reverse a Number.

  
var Reverse_Number = (N) => 
{
  let l=N.toString().length;
  let sum=0;
  let rem=0;
  for(let i=1;i<=l;i++){
    rem=N%10;
    sum=sum*10+rem;
    N=parseInt(N/10);
  }
  
  return sum;
};
